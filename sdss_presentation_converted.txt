\title Imaging Artifacts
\begin{Content}
\None
\begin{columns}[T]
\column{0.5\textwidth}
\textbf{Common imaging artifacts:}
\begin{itemize}
\item Diffraction spikes: Cross-shaped patterns around bright stars (from telescope spider vanes)
\item Saturation: Overexposed cores of bright stars bleed charge
\item Cosmic rays: Random bright pixels from high-energy particles
\item CCD defects: Dead columns, hot pixels
\end{itemize}

\title The Concept of Redshift (z)
\begin{Content}
\None
\begin{columns}
\column{0.5\textwidth}
\mosaic{2,2}{media_files/doppler.png,media_files/HRshift.png,media_files/Redshift.svg.png, media_files/screen_capture_20260405-212737.png}
\column{0.5\textwidth}
\textbf{Redshift formula}
$z = \frac{\lambda_{\text{observed}} - \lambda_{\text{rest}}}{\lambda_{\text{rest}}}$
\begin{itemize}
\item Universe expands → stretches light to longer wavelengths
\item \alert{Higher z = farther away = earlier universe}
\item SDSS measures z for millions of objects
\end{itemize}
